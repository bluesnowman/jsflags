//this is the class pattern these objects follow

// parent constructor function Person() {
// an "own" property
this.name = "Adam"; }
// a property added to the prototype Person.prototype.getName = function () {
return this.name; };
// inherit
var kid = object(Person.prototype);
typeof kid.getName; // "function", because it was in the prototype
typeof kid.name; // "undefined", because only the prototype was inherited
